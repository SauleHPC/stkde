\begin{frame}
  \frametitle{Space Time Kernel Density}
  
  \begin{block}{What is it?}
    \begin{itemize}
    \item Common way of visualizing events with time and place
      information
    \item Basically voxelize the space
    \item Give a value to each voxel that depends on the number of
      {\it neighboring} events to the voxel (with some kind of decay).
    \item Essentially a generalization of density maps (e.g.,
      population density)
    \end{itemize}
  \end{block}

  \begin{block}{What is it useful for?}
    \begin{itemize}
    \item Monitoring disease outbreak
    \item Political analysis
    \item Social media analysis
    \item Ornithology
    \end{itemize}
  \end{block}  
\end{frame}

\begin{frame}
  \frametitle{Space-Time Kernel Density Estimate Formally}

  \begin{columns}
    \column{.59\linewidth}

    \begin{block}{For a voxel $x,y,t$}
      $\hat{f} (x, y, t) = \frac{1}{n h_s^2 h_t} \sum_{i | d_i < h_s,
        t_i<h_t} k_s (\frac{x-x_i}{h_s},\frac{y-y_i}{h_s})
      k_t(\frac{t-t_i}{h_t})$
    
      $$k_s (u,v) = \frac{\pi}{2} (1-u)^2 (1-v)^2$$
      $$k_t (w) = \frac{3}{4} (1-w)^2$$

      $h_s$ is the spatial bandwidth

      $h_t$ is the temporal bandwidth

      $n$ is the number of points (events)
    \end{block}
    
    \column{.4\linewidth}

    \begin{block}{Each event radiates density}
    
      \includegraphics[width=\linewidth]{../../paper/icpp17/figures/stkde.pdf}
    \end{block}
  \end{columns}

  Similar to computing sums of radial basis functions which are typical in physics.
\end{frame}

\section{Sequential Algorithms}

\begin{frame}[fragile]
  \frametitle{Voxel-Based Algorithm \textsc{VB}}

  \begin{block}{Algorithm}
    \begin{algorithmic}
      \For{all voxels $s=(x,y,t)$}
      \State $sum = 0$
      \For{all points $i$ at $x_i, y_i, t_i$}
      \If{$\sqrt{(x_i-x)^2+(y_i-y)^2}< h_s$ and $|t_i-t| \leq h_t$}
      \State $sum += k_s(\frac{x-x_i}{h_s},\frac{y-y_i}{h_s}) k_t(\frac{t-t_i}{h_t})$
      \EndIf
      \EndFor
      \State $stkde[X][Y][T] = \frac{sum}{n h_s^2h_t}$
      \EndFor
    \end{algorithmic}
  \end{block}

  \begin{itemize}
  \item $\theta(G_x G_y G_t n)$ distance tests
  \item $\theta(n H_s^2 H_t )$ density values
  \item Complexity: $\theta(G_x G_y G_t n )$
  \end{itemize}

  But pleasingly parallel.    
\end{frame}

\begin{frame}
  \frametitle{Point-Based Algorithm \textsc{PB}}
  
  \begin{block}{Algorithm}
    \begin{algorithmic}
      \For{all voxels $s=(x,y,t)$}
        \State $stkde[X][Y][T] = 0$
      \EndFor
      %
      \For{each points $i$ at $x_i, y_i, t_i$}
        \For{$X_i - H_s\leq X \leq X_i+H_s$}
          \For{$Y_i - H_s\leq Y \leq Y_i+H_s$}
            \For{$T_i - T_s\leq T \leq T_i+H_s$}
              \If{$\sqrt{(x_i-x)^2+(y_i-y)^2}< h_s$ and $|t_i-t| \leq h_t$}
                \State $stkde[X][Y][T] += \frac{k_s(\frac{x-x_i}{h_s},\frac{y-y_i}{h_s}) k_t(\frac{t-t_i}{h_t})}{n h_s^2h_t}$
              \EndIf
            \EndFor
          \EndFor
        \EndFor
      \EndFor
    \end{algorithmic}
  \end{block}
    
  \begin{itemize}
  \item $\Theta(G_x G_y G_t)$ for memory initialization
  \item $\Theta(n H_s^2 H_t)$ density computations
  \item Complexity:  $\Theta(G_x G_y G_t + n H_s^2 H_t)$
  \item (Gain the $\theta(G_x G_y G_t n)$ distance tests from the voxel-based algorithm)
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Parallelism in STKDE is not trivial}

  \begin{block}{The problem}
    \begin{itemize}
    \item The algorithm is written with a \textbf{for all points} as its outer loop.
    \item But if the cylinder of two points intersect and the points are
      processed at the same time, there could be race condition in
      writing the \texttt{stkde} array
    \end{itemize}
  \end{block}

  \begin{block}{Naive solution}
    \begin{itemize}
    \item Lock the state of cells of \texttt{stkde} to avoid the race condition
    \item Or use atomics when updating \texttt{stkde}
    \item That causes $\Theta(n H_s^2 H_t)$ locks or atomics
    \end{itemize}
  \end{block}

  \pause

  \begin{block}{A better solution}
    \begin{itemize}
    \item Make sure intersecting cylinder are never processed at the same time.
    \item That is the rest of the presentation
    \end{itemize}  
  \end{block}
\end{frame}

\section{Domain-Based Parallelism}

\begin{frame}
  \frametitle{Domain Replication \textsc{PB-SYM-DR}}

  Each worker:
  \begin{itemize}
  \item Initializes its own memory buffer \texttt{stkde\_local}
  \item Processes some points in \texttt{stkde\_local} (with load balancing)
    \begin{itemize}
    \item So no race condition
    \end{itemize}
  \item Participates in reducing the many \texttt{stkde\_local} into \texttt{stkde}
  \end{itemize}
  
  \includegraphics[width=1.05\linewidth]{../../paper/icpp17/analysis/SYM-DR-speedup.pdf}  
\end{frame}

\begin{frame}
  \frametitle{Why is DR bad? Some instances have little computation!}

    \includegraphics[width=.8\linewidth]{../../paper/icpp17/analysis/seq_breakup.pdf}

    (and some run out of memory)
\end{frame}

\begin{frame}
  \frametitle{Domain Decomposition \textsc{PB-SYM-DD}}
  
  \begin{itemize}
  \item Decompose the voxel domain in $K \times K \times K$ subdomains
  \item Each worker processes different subdomains (load balanced on the subdomains)
    \begin{itemize}
      \item Each voxel is now processed by a unique thread, so no race condition
    \end{itemize}
  \end{itemize}

  \includegraphics[width=1.05\linewidth]{../../paper/icpp17/analysis/SYM-DD-speedup16.pdf}  
\end{frame}

\begin{frame}
  \frametitle{Optimal decompositions can be found using dynamic programming}

  \begin{equation}
    \begin{split}
  & Hier(X_{min}, X_{max}, Y_{min}, Y_{max}, T_{min}, T_{max}, P) = \min_{1 \leq p <P} (\\
\min_{X_{min}<x<X_{max}}  ( \max& Hier(X_{min}, x, Y_{min}, Y_{max}, T_{min}, T_{max}, p),\\
                         & Hier(x, X_{max}, Y_{min}, Y_{max}, T_{min}, T_{max}, P-p) ),\\
\min_{Y_{min}<y<Y_{max}}  ( \max & Hier(X_{min}, X_{max}, Y_{min}, y, T_{min}, T_{max}, p),\\
                          & Hier(X_{min}, X_{max}, y, Y_{max}, T_{min}, T_{max}, P-p) ),\\
\min_{T_{min}<t<T_{max}}  ( \max &Hier(X_{min}, X_{max}, Y_{min}, Y_{max}, T_{min}, t, p),\\
                    & Hier(X_{min}, X_{max}, Y_{min}, Y_{max}, t, T_{max}, P-p) ) )\\
    \end{split}
\end{equation}

\section{Point-Based Parallelism}

\begin{frame}
  \frametitle{Point decomposition \textsc{PB-SYM-PD}}

  \begin{itemize}
  \item Partition the points in a regular AxBxC grid such that each
    dimension is bigger than the bandwidth.
  \item For $(a,b,c) \in \{0,1\}^3$
    \begin{itemize}
    \item Do in parallel grids $2i+a,2j+b,2k+c, \forall i,j,k$
    \item Sync
      \begin{itemize}
      \item No two points closer than twice the bandwidth are processed
        simultaneously, so no simultaneous cylinder intersection.
      \end{itemize}
    \end{itemize}
  \end{itemize}
  
  \includegraphics[width=1.05\linewidth]{../../paper/icpp17/analysis/SYM-PD-speedup16.pdf}  
\end{frame}

\begin{frame}
  \frametitle{Why is this bad? Too many dependencies?}

  Since all $2i, 2j, 2k$ are done before any $2i+1, 2j, 2k$, there is
  a forced precedence of $0, 0, 0$ over $3, 0, 0$. But they are not
  dependent.

  This does coloring, instead of doing scheduling.
  \begin{center}
    \includegraphics[width=.5\linewidth]{../../paper/icpp17/figures/PB-SYM-PD-graph.pdf}
  \end{center}
  
  Building the graph from a coloring is simple (and easily expressed in OpenMP 4.0).
\end{frame}

\begin{frame}
  \frametitle{A better coloring with \textsc{PD-SYM-PD-SCHED}}

  We don't need to color subdomains to minimize the number of colors.

  We need a coloring that minimizes the longest chain in the implied graph.

  Heuristic: greedily color subdomains in highest number of points first.
   
  \includegraphics[width=\linewidth]{../../paper/icpp17/analysis/SYM-PD-COLOR-SCHED-speedup16.pdf}

  (If you don't have a good eye, it is a bit better than before)
\end{frame}

\begin{frame}
  \frametitle{Why is it bad? Still too long critical path!}
  \begin{center}
    \includegraphics[width=.6\linewidth]{../../paper/icpp17/analysis/PB-SYM-PD-imbalance.pdf}
  \end{center}
    
  How hard is the coloring/edge-orientation problem to minimize critical path?
  \begin{itemize}
  \item NP-hard in general graph (harder than coloring)
  \item Trivial on chains
  \item Other?
  \end{itemize}

  How good is the heuristic?

  I need to think more about this...
\end{frame}

\begin{frame}
  \frametitle{Parallelize long path \textsc{PB-SYM-PD-SCHED-REP}}

  One can replicate a subdomain and get perfect work parallelism. (at
  the expense of some memory initialization and reduction.)

  
  Heuristic: for all path longer than $\frac{n}{2P}$, add one copy to all tasks on the path

  \includegraphics[width=\linewidth]{../../paper/icpp17/analysis/SYM-PD-COLOR-REP-speedup16.pdf}  

  (This sounds like moldable DAG scheduling.)

\end{frame}
